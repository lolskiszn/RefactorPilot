class GitNativeEngine:
    """Use libgit2/pygit2 for performance and atomicity"""
    
    def init(self, repo_path: str):
        self.repo = pygit2.Repository(repo_path)
        self.index = self.repo.index
    
    def stage_migration(self, migration_id: str, changes: List[ChangeUnit]):
        """Atomic staging with metadata embedding"""
        branch_name = f"refactorpilot/{migration_id}"
        
        # Create branch from current HEAD
        parent = self.repo.head.target
        self.repo.create_branch(branch_name, self.repo[parent])
        
        # Apply changes in dependency order
        for batch in self._topological_batch(changes):
            for change in batch:
                self._apply_change(change)
            
            # Commit per batch with traceability
            self._commit_batch(batch, migration_id)
        
        # Embed migration metadata in git notes
        self._attach_metadata(migration_id, changes)
        
        return MigrationBranch(
            name=branch_name,
            commits=self._get_commits(branch_name),
metadata=migration_id
        )
    
    def atomic_rollback(self, migration_id: str):
        """Revert all changes or nothing—distributed systems style"""
        # Find all branches with this migration
        branches = self._find_migration_branches(migration_id)
        
        for branch in branches:
            # Revert commits in reverse order
            for commit in reversed(branch.commits):
                self._revert_commit(commit)
            
            # Delete branch
            self.repo.delete_branch(branch.name)
        
        # Clean metadata
        self._cleanup_metadata(migration_id)

@dataclass
class RefactorIR:
    """
    Multi-layer IR. Each layer adds semantic richness.
    Lower layers are language-agnostic. Higher layers capture intent.
    """
    
    # Layer 0: Syntactic structure (from Tree-sitter)
    syntax_tree: SyntaxNode  # Universal AST
    
    # Layer 1: Control & Data Flow (SSA form)
    control_flow: DiGraph    # Basic blocks, edges
    data_flow: DiGraph       # Def-use chains, reaching definitions
    
    # Layer 2: Semantic entities (functions, types, interfaces)
    symbols: SymbolTable     # Scoped identifiers
    types: TypeGraph         # Type constraints, inference
    
    # Layer 3: Architectural concepts (APIs, boundaries, contracts)
    api_surface: APIGraph    # Endpoints, methods, schemas
    boundaries: BoundaryGraph  # Cross-language call sites
    
    # Layer 4: Intent annotations (human or inferred)
    patterns: List[DetectedPattern]  # "This is a repository pattern"
    contracts: List[Contract]       # Pre/post conditions

class IRConstructor:
    def construct(self, source_files: List[SourceFile]) -> RefactorIR:
        """Multi-pass construction with cross-language linking"""
        
        # Pass 1: Parse each file to syntax tree
        syntax_trees = {
            f.path: self.parser.parse(f.content, f.language)
            for f in source_files
        }
        
        # Pass 2: Build per-language semantic models
        semantic_models = {}
        for path, tree in syntax_trees.items():
            lang = self._detect_language(path)
model = self._language_analyzers[lang].analyze(tree)
            semantic_models[path] = model
        
        # Pass 3: Cross-language boundary detection
        # This is the magic: find where Go calls Python via HTTP/gRPC/FFI
        boundaries = self._detect_boundaries(semantic_models)
        
        # Pass 4: Unified property graph construction
        return self._build_unified_graph(syntax_trees, semantic_models, boundaries)
    
    def _detect_boundaries(self, models: Dict[str, SemanticModel]) -> List[Boundary]:
        """
        Find cross-language links. Examples:
        - Go struct <-> JSON <-> Python dataclass
        - Go HTTP handler <-> OpenAPI spec <-> Python requests call
        - gRPC proto <-> Go generated code <-> Python generated code
        """
        boundaries = []
        
        # Strategy 1: Schema-based (OpenAPI, proto, GraphQL)
        schemas = self._extract_schemas(models)
        for schema in schemas:
            implementations = self._find_implementations(schema, models)
            boundaries.append(SchemaBoundary(schema, implementations))
        
        # Strategy 2: Convention-based (REST endpoints, naming patterns)
        endpoints = self._detect_rest_endpoints(models)
        for endpoint in endpoints:
            server = self._find_server(endpoint, models)
            clients = self._find_clients(endpoint, models)
            boundaries.append(APIBoundary(endpoint, server, clients))
        
        # Strategy 3: Explicit linking (FFI, cgo, pybind11)
        ffi_calls = self._detect_ffi_calls(models)
        for call in ffi_calls:
            boundaries.append(FFIBoundary(call))
        
        return boundaries

class IncrementalIR:
    """Only re-analyze changed files. Critical for 100k+ file repos."""
    
    def init(self, cache_dir: str):
        self.cache = AnalysisCache(cache_dir)
        self.hasher = ContentHasher()
    
    def analyze(self, files: List[SourceFile], previous_ir: Optional[RefactorIR]) -> RefactorIR:
        # Hash all files
        current_hashes = {f.path: self.hasher.hash(f.content) for f in files}
        
        # Find changed files
        changed_files = []
        unchanged_files = []
        for f in files:
            cached = self.cache.get_hash(f.path)
            if cached != current_hashes[f.path]:
                changed_files.append(f)
            else:
                unchanged_files.append(f)
        
        # Load cached IR for unchanged files
        cached_ir = self.cache.load_ir(unchanged_files)
        
        # Re-analyze only changed files
        new_ir = self._analyze_changed(changed_files)
        
        # Merge: preserve boundaries that don't involve changed files
        # Recalculate boundaries that touch changed files
        merged = self._merge_ir(cached_ir, new_ir, previous_ir)
        
        # Update cache
        self.cache.store(current_hashes, merged)
        
        return merged

class EquivalenceProver:
    """
    Prove that transformation preserves behavior.
    Hybrid approach: static analysis + dynamic verification.
    """
    
    def prove(self, original: IRFragment, transformed: IRFragment) -> ProofResult:
        # Strategy 1: Syntactic equivalence (exact same operations)
        if self._syntactic_equiv(original, transformed):
            return ProofResult(
                confidence=1.0,
                method="syntactic",
                explanation="Identical control flow and data flow"
            )
        
        # Strategy 2: Semantic equivalence (different syntax, same semantics)
        if self._semantic_equiv(original, transformed):
            return ProofResult(
                confidence=0.95,
                method="semantic",
                explanation="Equivalent up to refactoring laws"
            )
        
        # Strategy 3: Contract-based (pre/post conditions preserved)
        contracts = self._extract_contracts(original)
        if self._verify_contracts_preserved(transformed, contracts):
            return ProofResult(
                confidence=0.90,
                method="contract",
                explanation="Pre/post conditions preserved"
            )
        
        # Strategy 4: Differential testing (run both, compare outputs)
        test_result = self._differential_test(original, transformed)
        if test_result.pass_rate > 0.99:
            return ProofResult(
                confidence=0.85,
                method="differential",
                explanation=f"Passed {test_result.pass_rate*100}% of tests"
            )
        
        # Strategy 5: Human verification required
        return ProofResult(
            confidence=0.0,
            method="manual",
            explanation="Cannot automatically prove equivalence"
        )
    
    def _semantic_equiv(self, a: IRFragment, b: IRFragment) -> bool:
"""Check if two fragments are semantically equivalent"""
        # Normalize both to SSA form
        a_ssa = self._to_ssa(a)
        b_ssa = self._to_ssa(b)
        
        # Check if data flow graphs are isomorphic
        return self._graph_isomorphism(a_ssa.data_flow, b_ssa.data_flow)

class MigrationOrchestrator:
    """Coordinate migrations across hundreds of repositories"""
    
    def init(self, manifest: OrganizationManifest):
        self.manifest = manifest  # All repos, dependencies, teams
        self.scheduler = DependencyScheduler()
    
    def plan_organization_migration(
        self,
        pattern: MigrationPattern,
        target_scope: MigrationScope
    ) -> OrganizationMigrationPlan:
        """
        Plan migration across the entire org.
        Respects service dependencies, team boundaries, deployment order.
        """
        
        # Build global dependency graph
        services = self.manifest.get_services(target_scope)
        dep_graph = self._build_dependency_graph(services)
        
        # Topological sort: migrate leaves first (no dependents)
        migration_order = list(nx.topological_sort(dep_graph.reverse()))
        
        # Batch by team and risk level
        batches = []
        for service in migration_order:
            team = self.manifest.get_team(service)
            risk = self._assess_risk(service, pattern)
            
            batches.append(MigrationBatch(
                service=service,
                team=team,
                risk_level=risk,
                estimated_duration=self._estimate(service, pattern),
                rollback_time=self._estimate_rollback(service)
            ))
        
        # Schedule with constraints
        schedule = self.scheduler.schedule(
            batches,
            constraints=[
                MaxConcurrentPerTeam(2),  # Don't overwhelm teams
                MaxRiskPerWeek("medium"),  # Limit blast radius
                DependencyOrder(),  # Don't migrate before dependencies
                MaintenanceWindow("tue-thu 9-17")  # Business hours
            ]
        )
        
        return OrganizationMigrationPlan(
            schedule=schedule,
            total_estimated_duration=sum(b.duration for b in batches),
            risk_mitigation=self._risk_mitigation(batches),
            rollback_strategy=self._rollback_strategy(batches)
        )
    
    def execute_batch(self, batch: MigrationBatch) -> BatchResult:
        """Execute one batch with full observability"""
        results = []
        
        for service in batch.services:
            # Create PR via GitHub/GitLab API
            pr = self._create_migration_pr(service, batch.pattern)
            
            # Wait for CI (with timeout)
            ci_result = self._wait_for_ci(pr, timeout=timedelta(hours=2))
            
            if ci_result.success:
                # Auto-merge if confidence high and policy allows
                if batch.risk_level == "low" and self.manifest.auto_merge_enabled:
                    self._merge_pr(pr)
                    results.append(ServiceResult(service, "merged", pr.url))
                else:
                    # Request human review
                    self._notify_team(batch.team, pr)
                    results.append(ServiceResult(service, "pending_review", pr.url))
            else:
                # Rollback this service, continue with others
                self._rollback_pr(pr)
                results.append(ServiceResult(service, "failed", ci_result.logs))
        
        return BatchResult(results)

class PolicyEngine:

    def check_compliance(self, repo: Repository) -> ComplianceReport:
        """Check if repo follows mandated patterns"""
        policies = self._load_policies(repo.organization)
        
        violations = []
        for policy in policies:
            if not policy.check(repo):
                violations.append(Violation(
                    policy=policy,
                    severity=policy.severity,
                    auto_fixable=policy.can_auto_fix,
                    suggested_migration=self._suggest_fix(policy, repo)
                ))
        
        return ComplianceReport(violations)
    
    def enforce_policy(self, policy: Policy, scope: OrganizationScope):
